def load_embedding_model(model_name: str = 'sentence-transformers/all-MiniLM-L6-v2',
                        cache_dir: str = './storage/models') -> SentenceTransformer:
    os.makedirs(cache_dir, exist_ok=True)
    local_model_path = os.path.join(cache_dir, model_name.replace('/', '_'))
    try:
        if os.path.exists(local_model_path) and os.listdir(local_model_path):
            model = SentenceTransformer(local_model_path)
            return model
    except Exception as e:
        print(f"Failed to load local model: {e}")
    try:
        model = SentenceTransformer(model_name)
        model.save(local_model_path)
        return model
    except Exception as e:
        raise RuntimeError(f"Unable to load embedding model {model_name}: {e}")

class AdaptiveEchoMemoryCore:
    """
    Core memory store with dual-stack architecture.
    - Raw stack: stores all trajectories (for bootstrap fallback)
    - Core stack: stores refined, high-quality summaries from successful tasks
    Implements incremental embedding, multi-field indexing with phase-aware weighting,

    def __init__(self, config: dict, embedding_model: SentenceTransformer):

        self.merge_threshold = config.get("merge_threshold", 0.80)

        self.core_memories: Dict[str, MemoryEntry] = {}  # id -> entry
        self.core_embeddings: Optional[np.ndarray] = None  # (N, embedding_dim)
        self.core_ids: List[str] = []
        self.embedding_dim: Optional[int] = None
        
        # Raw trajectory storage
        self.raw_entries: Dict[str, RawTrajectoryEntry] = {}
        self.raw_embeddings: Optional[np.ndarray] = None
        self.raw_ids: List[str] = []
        
        # Staging for incremental indexing
        self.staging_core: List[MemoryEntry] = []
        self.staging_raw: List[RawTrajectoryEntry] = []
        self.staging_threshold = config.get("staging_threshold", 10)

        self.total_retrievals = 0

        # Load existing data if any
        self._load_databases()
    
    def _load_databases(self):
        """Load existing databases from disk"""
        for db_path, storage_dict, attr_name in [
            (self.core_db_path, self.core_memories, "core"),
            (self.raw_db_path, self.raw_entries, "raw")
        ]:
            if os.path.exists(db_path):
                try:
                    with open(db_path, 'r', encoding='utf-8') as f:
                        data = json.load(f)
                    for item in data:
                        entry = self._dict_to_entry(item, attr_name)
                        if entry:
                            storage_dict[entry.id] = entry
                except Exception as e:
                    print(f"Error loading {db_path}: {e}")
        self._rebuild_all_embeddings()
    
    def _dict_to_entry(self, item: dict, attr_name: str):
        """Convert dict to MemoryEntry or RawTrajectoryEntry"""
        if attr_name == "core":
            return MemoryEntry(
                id=item.get("id", str(uuid.uuid4())),
                query=item.get("query", ""),
                planning=item.get("planning", ""),
                search_planning=item.get("search_planning", ""),
                experience=item.get("experience", ""),
                search_experience=item.get("search_experience", ""),
                timestamp=item.get("timestamp", ""),
                success=item.get("success", False),
                reuse_count=item.get("reuse_count", 0),
                success_count=item.get("success_count", 0),
                utility_score=item.get("utility_score", 0.0),
                embedding=None  # will be recomputed
            )
        else:  # raw
            return RawTrajectoryEntry(
                id=item.get("id", str(uuid.uuid4())),
                query=item.get("query", ""),
                trajectory_text=item.get("trajectory_text", ""),
                result=item.get("result", None),
                timestamp=item.get("timestamp", ""),
                success=item.get("success", False)
            )
    
    def _rebuild_all_embeddings(self):
        """Rebuild all embeddings from scratch (used on load)"""
        # Core memories
        if self.core_memories:
            entries_list = list(self.core_memories.values())
            texts = []
            for e in entries_list:
                combined = f"{e.query} {e.planning} {e.search_planning} {e.experience} {e.search_experience}"
                texts.append(combined)
            if texts:
                embeddings = self.embedding_model.encode(texts, convert_to_numpy=True)
                self.core_embeddings = embeddings
                self.core_ids = [e.id for e in entries_list]
                # Store embeddings in entries for incremental use
                for i, e in enumerate(entries_list):
                    e.embedding = embeddings[i]
                self.embedding_dim = embeddings.shape[1]
        else:
            self.core_embeddings = None
            self.core_ids = []
        
        # Raw trajectories
        if self.raw_entries:
            entries_list = list(self.raw_entries.values())
            texts = [f"{e.query} {e.trajectory_text}" for e in entries_list]
            if texts:
                raw_embeddings = self.embedding_model.encode(texts, convert_to_numpy=True)
                self.raw_embeddings = raw_embeddings
                self.raw_ids = [e.id for e in entries_list]
                if self.embedding_dim is None:
                    self.embedding_dim = raw_embeddings.shape[1]
        else:
            self.raw_embeddings = None
            self.raw_ids = []
    
    def _compute_query_embedding(self, query: str) -> np.ndarray:
        """Compute embedding for a query string"""
        return self.embedding_model.encode([query], convert_to_numpy=True)[0]
    
    def _compute_complexity(self, query: str) -> float:
        """Compute query complexity based on length and entity-like words"""
        # Simple heuristic: length + number of capitalized words
        words = query.split()
        length_score = min(len(words) / 10.0, 1.0)
        capital_count = sum(1 for w in words if w[0].isupper() if len(w) > 0)
        entity_score = min(capital_count / 3.0, 1.0)
        return 0.5 * length_score + 0.5 * entity_score
    
    def _dynamic_top_k(self, complexity: float) -> int:
        """Adaptive top-k based on query complexity"""
        # Complexity in [0,2], map to top_k in [1, 2*top_k_base]
        max_k = 2 * self.top_k_base
        k = int(self.top_k_base + (max_k - self.top_k_base) * complexity)
        return max(1, min(k, max_k))
    
    def _search_core(self, query: str, phase: MemoryStatus, top_k: int) -> List[MemoryEntry]:
        """Search core memories with phase-aware multi-field weighting"""
        if not self.core_embeddings or len(self.core_ids) == 0:
            return []
        
        query_emb = self._compute_query_embedding(query)
        query_emb = query_emb.reshape(1, -1)
        similarities = cosine_similarity(query_emb, self.core_embeddings)[0]
        
        # Phase-aware field weighting not fully implemented but we could re-rank by
        # similarity to specific fields later. For now, use overall similarity.
        # Could be extended to compute separate embeddings for planning/experience fields.
        top_indices = similarities.argsort()[-top_k:][::-1]
        results = []
        for idx in top_indices:
            entry_id = self.core_ids[idx]
            entry = self.core_memories[entry_id]
            entry.reuse_count += 1
            results.append(entry)
        return results
    
    def _search_raw(self, query: str, top_k: int) -> List[RawTrajectoryEntry]:
        """Search raw trajectory database"""
        if not self.raw_embeddings or len(self.raw_ids) == 0:
            return []
        query_emb = self._compute_query_embedding(query).reshape(1, -1)
        similarities = cosine_similarity(query_emb, self.raw_embeddings)[0]
        top_indices = similarities.argsort()[-top_k:][::-1]
        results = []
        for idx in top_indices:
            entry_id = self.raw_ids[idx]
            results.append(self.raw_entries[entry_id])
        return results
    
    def retrieve(self, query: str, phase: MemoryStatus) -> Tuple[List[Any], str]:
        """Main retrieval method. Returns (entries, source) where source is 'core' or 'raw'"""
        complexity = self._compute_complexity(query)
        top_k = self._dynamic_top_k(complexity)
        
        # Compute phase weights for possible field-level reranking (simplified: not implemented)
        # For now, use overall embedding similarity but could multiply by field weights.
        
        # Try core memories first
        core_results = self._search_core(query, phase, top_k)
        if core_results:
            return core_results, "core"
        
        # Fallback to raw trajectories
        raw_results = self._search_raw(query, top_k)
        if raw_results:
            return raw_results, "raw"
        
        return [], "none"
    
    def add_core_memory(self, entry: MemoryEntry):
        """Add a refined memory to core database (with dedup and merge)"""
        # Check for redundancy
        if self.core_embeddings is not None and len(self.core_ids) > 0:
            new_emb = self._compute_query_embedding(entry.query)
            # Check similarity with existing entries
            similarities = cosine_similarity([new_emb], self.core_embeddings)[0]
            max_sim = np.max(similarities) if len(similarities) > 0 else 0.0
            if max_sim > self.similarity_threshold:
                # Duplicate: skip
                print(f"Duplicate memory (sim={max_sim:.3f}), skipping.")
                return
            elif max_sim > self.merge_threshold:
                # Merge with the most similar entry
                best_idx = np.argmax(similarities)
                best_id = self.core_ids[best_idx]
                best_entry = self.core_memories[best_id]
                # Merge by averaging timestamps, updating content with longer version
                best_entry.planning = max(best_entry.planning, entry.planning, key=len)
                best_entry.search_planning = max(best_entry.search_planning, entry.search_planning, key=len)
                best_entry.experience = max(best_entry.experience, entry.experience, key=len)
                best_entry.search_experience = max(best_entry.search_experience, entry.search_experience, key=len)
                best_entry.timestamp = max(best_entry.timestamp, entry.timestamp)  # keep newer
                # Recompute embedding for merged entry
                combined = f"{best_entry.query} {best_entry.planning} {best_entry.search_planning} {best_entry.experience} {best_entry.search_experience}"
                new_merged_emb = self._compute_query_embedding(combined)
                best_entry.embedding = new_merged_emb
                # Update embedding matrix
                self.core_embeddings[best_idx] = new_merged_emb
                print(f"Merged memory into {best_id}")
                return
        
        # No duplicate: add to staging
        self.staging_core.append(entry)
        if len(self.staging_core) >= self.staging_threshold:
            self._flush_core_staging()
    
    def _flush_core_staging(self):
        """Incrementally add staged core memories to main index"""
        if not self.staging_core:
            return
        new_entries = self.staging_core
        self.staging_core = []
        
        # Compute embeddings for new entries
        texts = [f"{e.query} {e.planning} {e.search_planning} {e.experience} {e.search_experience}" for e in new_entries]
        new_embeddings = self.embedding_model.encode(texts, convert_to_numpy=True)
        for i, e in enumerate(new_entries):
            e.embedding = new_embeddings[i]
        
        # Add to main arrays
        if self.core_embeddings is None:
            self.core_embeddings = new_embeddings
            self.core_ids = [e.id for e in new_entries]
        else:
            self.core_embeddings = np.vstack([self.core_embeddings, new_embeddings])
            self.core_ids.extend([e.id for e in new_entries])
        
        for e in new_entries:
            self.core_memories[e.id] = e
        
        self.embedding_dim = self.core_embeddings.shape[1]
        self._save_core_db()
        print(f"Flushed {len(new_entries)} core memories to index.")
    
    def add_raw_trajectory(self, entry: RawTrajectoryEntry):
        """Add raw trajectory to staging, then flush when threshold reached"""
        self.staging_raw.append(entry)
        if len(self.staging_raw) >= self.staging_threshold:
            self._flush_raw_staging()
    
    def _flush_raw_staging(self):
        """Incrementally add staged raw entries"""
        if not self.staging_raw:
            return
        new_entries = self.staging_raw
        self.staging_raw = []
        
        texts = [f"{e.query} {e.trajectory_text}" for e in new_entries]

        else:
            self.raw_embeddings = np.vstack([self.raw_embeddings, new_embeddings])
            self.raw_ids.extend([e.id for e in new_entries])
        
        for e in new_entries:

        self._save_raw_db()
    
    def _save_core_db(self):

        data = []
        for e in self.core_memories.values():
            data.append({

                "query": e.query,

                "search_planning": e.search_planning,
                "experience": e.experience,

            })
        os.makedirs(os.path.dirname(self.core_db_path), exist_ok=True)
        with open(self.core_db_path, 'w', encoding='utf-8') as f:
            json.dump(data, f, indent=2, ensure_ascii=False)
    
    def _save_raw_db(self):
        """Serialize raw trajectories to JSON"""
        data = []
        for e in self.raw_entries.values():
            data.append({
                "id": e.id,
                "query": e.query,
                "trajectory_text": e.trajectory_text,
                "result": e.result,
                "timestamp": e.timestamp,
                "success": e.success
            })
        os.makedirs(os.path.dirname(self.raw_db_path), exist_ok=True)
        with open(self.raw_db_path, 'w', encoding='utf-8') as f:
            json.dump(data, f, indent=2, ensure_ascii=False)
    
    def record_success(self, memory_id: str, is_success: bool):
        """Update utility score for a core memory after usage"""
        if memory_id in self.core_memories:
            entry = self.core_memories[memory_id]
            if is_success:
                entry.success_count += 1
            entry.reuse_count += 1
            # Simple utility: success ratio, with recency (not yet implemented)
            if entry.reuse_count > 0:
                entry.utility_score = entry.success_count / entry.reuse_count
            # Trigger prune check
            self.total_retrievals += 1
            if self.total_retrievals % self.prune_interval == 0:
                self._prune_low_utility()
    
    def _prune_low_utility(self):
        """Remove memories with utility below threshold, keeping top N"""
        if len(self.core_ids) <= self.max_core_size:
            return
        # Sort by utility_score ascending, remove bottom 20%
        entries_with_score = [(e.utility_score, e.id) for e in self.core_memories.values()]
        entries_with_score.sort(key=lambda x: x[0])
        num_to_remove = int(len(entries_with_score) * 0.2)
        ids_to_remove = set(eid for _, eid in entries_with_score[:num_to_remove])
        
        # Remove from dict and rebuild embeddings
        removed = 0
        for eid in ids_to_remove:
            if eid in self.core_memories:
                del self.core_memories[eid]
                removed += 1
        if removed > 0:
            print(f"Pruned {removed} low-utility core memories.")
            self._rebuild_all_embeddings()
            self._save_core_db()
    
    def close(self):
        """Flush any remaining staging and save"""
        self._flush_core_staging()
        self._flush_raw_staging()
        self._save_core_db()
        self._save_raw_db()

    def initialize(self) -> bool:
        try:

            )

            core_config = {
                "raw_db_path": self.raw_db_path,
                "core_db_path": self.core_db_path,

                "phase_weights_in": self.phase_weights_in
            }
            self.core = AdaptiveEchoMemoryCore(core_config, self.embedding_model)
            self.initialized = True
            print(f"AdaptiveEchoMemory initialized: core entries {len(self.core.core_ids)}, raw entries {len(self.core.raw_ids)}")

    def _extract_query_refinement(self, request: MemoryRequest) -> str:
        """Use LLM to extract key search terms from the query (if model available)"""
        if not self.model:
            return request.query
        prompt = f"""Extract core concepts and key terms from the following query for effective memory retrieval. Return only the refined query text (no extra explanation).

Query: {request.query}

Refined query:"""
        messages = [{"role": "user", "content": [{"type": "text", "text": prompt}]}]
        try:
            response = self.model(messages)
            refined = getattr(response, "content", str(response)).strip()
            return refined if refined else request.query
        except Exception as e:
            print(f"Error in query refinement: {e}")
            return request.query
    
    def _synthesize_guidance_from_core(self, entries: List[MemoryEntry], query: str, phase: MemoryStatus) -> str:
        """Synthesize guidance text from core memory entries using LLM or direct concatenation"""
        if not self.model:
            # Fallback: combine planning and experience based on phase
            guidance_parts = []
            for e in entries:
                if phase == MemoryStatus.BEGIN:
                    parts = [e.planning, e.search_planning]
                else:
                    parts = [e.experience, e.search_experience]
                guidance_parts.extend([p for p in parts if p])
            return " | ".join(guidance_parts) if guidance_parts else "No relevant guidance found."
        
        # Use LLM to synthesize
        fields = []
        for i, e in enumerate(entries):
            if phase == MemoryStatus.BEGIN:
                fields.append(f"Entry {i+1}\nPlanning: {e.planning}\nSearch Planning: {e.search_planning}")
            else:
                fields.append(f"Entry {i+1}\nExperience: {e.experience}\nSearch Experience: {e.search_experience}")
        
        content = "\n\n".join(fields)
        prompt = f"""You are an expert AI agent mentor. Based on the following retrieved memories, synthesize concise, actionable guidance for the current task.

Current Task: {query}
Phase: {"Initial planning" if phase == MemoryStatus.BEGIN else "Execution phase"}

Retrieved Memories:
{content}

Provide 2-3 specific, actionable suggestions that will help the agent succeed. Use gentle, suggestive language. No markdown, no headings, just the suggestions as plain text numbered list.

Suggestions:"""
        messages = [{"role": "user", "content": [{"type": "text", "text": prompt}]}]
        try:
            response = self.model(messages)
            guidance = getattr(response, "content", str(response)).strip()
            return guidance if guidance else "No specific guidance generated."
        except Exception as e:
            print(f"Error synthesizing guidance: {e}")
            return "No guidance available."
    
    def _synthesize_guidance_from_raw(self, entries: List[RawTrajectoryEntry], query: str) -> str:
        """Synthesize guidance from raw trajectory entries (bootstrap mode)"""
        if not self.model:
            # Fallback: combine trajectory_text
            texts = [e.trajectory_text[:500] for e in entries]  # limit length
            return "Raw trajectory guidance (bootstrap): " + " | ".join(texts)
        
        # Use LLM to extract patterns from raw trajectories
        content = "\n\n".join([f"Entry {i+1} Query: {e.query}\nTrajectory: {e.trajectory_text[:1000]}" for i,e in enumerate(entries)])
        prompt = f"""You are an AI agent learning from past attempts (including failures). Extract useful patterns, strategies, and pitfalls from the following raw trajectory logs to help with the current task.

Current Task: {query}

Past Trajectories:
{content}

Provide 2-3 concise, actionable insights (use gentle language). Output as numbered list.

Insights:"""
        messages = [{"role": "user", "content": [{"type": "text", "text": prompt}]}]
        try:
            response = self.model(messages)
            guidance = getattr(response, "content", str(response)).strip()
            return guidance if guidance else "No insights extracted."
        except Exception as e:
            print(f"Error synthesizing raw guidance: {e}")
            return "No guidance available."
    
    def provide_memory(self, request: MemoryRequest) -> MemoryResponse:
        if not self.initialized or not self.core:
            return MemoryResponse(
                memories=[], memory_type=self.memory_type, total_count=0,
                request_id=str(uuid.uuid4())
            )
        
        try:
            refined_query = self._extract_query_refinement(request)
            phase = request.status
            
            # Retrieve from core or raw
            entries, source = self.core.retrieve(refined_query, phase)
            
            if not entries:
                return MemoryResponse(
                    memories=[], memory_type=self.memory_type, total_count=0,
                    request_id=str(uuid.uuid4())
                )
            
            # Synthesize guidance based on source
            if source == "core":
                # entries are MemoryEntry
                guidance = self._synthesize_guidance_from_core(entries, request.query, phase)
            else:
                # entries are RawTrajectoryEntry
                guidance = self._synthesize_guidance_from_raw(entries, request.query)
            
            # Create memory item
            memory_item = MemoryItem(
                id=f"echo_{uuid.uuid4()}",
                content=guidance,
                metadata={
                    "source": source,
                    "num_entries": len(entries),
                    "refined_query": refined_query,
                    "phase": phase.value,
                    "entry_ids": [e.id for e in entries]
                },
                score=1.0  # placeholder
            )
            
            return MemoryResponse(
                memories=[memory_item],
                memory_type=self.memory_type,
                total_count=1,
                request_id=str(uuid.uuid4())
            )
        except Exception as e:
            print(f"Error providing memory: {e}")
            return MemoryResponse(
                memories=[], memory_type=self.memory_type, total_count=0,
                request_id=str(uuid.uuid4())
            )
    
    def take_in_memory(self, trajectory_data: TrajectoryData) -> tuple[bool, str]:
        """Ingest trajectory: always store raw; if successful, also store refined core memory"""
        if not self.initialized or not self.core:
            return False, "Memory system not initialized."
        
        try:
            # Extract basic info
            query = trajectory_data.query
            result = trajectory_data.result
            metadata = trajectory_data.metadata or {}
            is_success = metadata.get("is_correct", False) or metadata.get("success", False)
            
            # Format trajectory as text
            trajectory_text = self._format_trajectory(trajectory_data)
            
            # 1. Always store raw trajectory
            raw_entry = RawTrajectoryEntry(
                id=str(uuid.uuid4()),
                query=query,
                trajectory_text=trajectory_text,
                result=str(result) if result else None,
                timestamp=datetime.now().isoformat(),
                success=is_success
            )
            self.core.add_raw_trajectory(raw_entry)
            
            # 2. If successful, attempt to create refined core memory using LLM
            if is_success and self.model:
                # Use LLM to extract structured memory
                refined = self._extract_refined_memory(trajectory_data)
                if refined:
                    core_entry = MemoryEntry(
                        id=str(uuid.uuid4()),
                        query=query,
                        planning=refined.get("agent_planning", ""),
                        search_planning=refined.get("search_agent_planning", ""),
                        experience=refined.get("agent_experience", ""),
                        search_experience=refined.get("search_agent_experience", ""),
                        timestamp=datetime.now().isoformat(),
                        success=True,
                        reuse_count=0,
                        success_count=0,
                        utility_score=0.0
                    )
                    self.core.add_core_memory(core_entry)
                    self.core.record_success(core_entry.id, True)  # initial success
                    return True, f"Stored raw and refined core memory for '{query}'"
            
            # If not successful or no model, still stored raw
            return True, f"Stored raw trajectory for '{query}' (core memory skipped: success={is_success})"
        
        except Exception as e:
            print(f"Error taking in memory: {e}")
            return False, f"Error: {e}"
    
    def _format_trajectory(self, trajectory_data: TrajectoryData) -> str:
        """Convert trajectory data to a concatenated text string"""
        if not trajectory_data.trajectory:
            return "No trajectory steps."
        parts = [f"Query: {trajectory_data.query}"]
        for i, step in enumerate(trajectory_data.trajectory):
            step_type = step.get("type", "step")
            content = step.get("content", "")
            parts.append(f"Step {i+1} ({step_type}): {content}")
        if trajectory_data.result:
            parts.append(f"Result: {trajectory_data.result}")
        return "\n".join(parts)
    
    def _extract_refined_memory(self, trajectory_data: TrajectoryData) -> Optional[Dict[str, str]]:
        """Use LLM to extract structured memory from successful trajectory"""
        if not self.model:
            return None
        trajectory_text = self._format_trajectory(trajectory_data)
        prompt = f"""You are an expert AI agent trainer analyzing a successful task execution. Extract structured memory components that capture the strategic thinking and methodologies used.

Question: {trajectory_data.query}

Execution Trajectory:
{trajectory_text}

Final Result: {trajectory_data.result if trajectory_data.result else "Success"}

Extract the following four fields in JSON format. Each field must contain substantial, actionable content (at least 2-3 sentences). Focus on specific techniques, tool choices, and reasoning patterns.

JSON format:
{{
    "agent_planning": "Detailed planning approach including decision points and tool selection",
    "search_agent_planning": "Search strategy and query formulation techniques",
    "agent_experience": "Key lessons learned and best practices",
    "search_agent_experience": "Search-specific insights and validation methods"
}}

Return only the JSON object, no other text."""

        messages = [{"role": "user", "content": [{"type": "text", "text": prompt}]}]
        try:
            response = self.model(messages)
            response_text = getattr(response, "content", str(response)).strip()
            # Extract JSON block
            import re
            json_match = re.search(r'\{.*\}', response_text, re.DOTALL)
            if json_match:
                json_str = json_match.group(0)
                memory = json.loads(json_str)
                required = ["agent_planning", "search_agent_planning", "agent_experience", "search_agent_experience"]
                if all(k in memory and len(memory[k].strip()) >= 50 for k in required):
                    return memory
                else:
                    print(f"Memory fields too short or missing: {memory.keys()}")
            else:
                print(f"No JSON found in response: {response_text[:200]}")
        except Exception as e:
            print(f"Error extracting refined memory: {e}")
        return None
    
    def close(self):
        if self.core:
            self.core.close()
        super().close()
